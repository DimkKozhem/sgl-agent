async def create_task(self, request_data: Dict) -> str: # Проверка переполнения очереди if self.max_queue_size > 0 and len(self.tasks) >= self.max_queue_size: self.error_stats["queue_full_errors"] += 1 raise HTTPException( status_code=413, detail=f"Queue full: {len(self.tasks)}/{self.max_queue_size}" ) # Генерация UUID task_id = str(uuid.uuid4()) # Создание объекта задачи task = Task( task_id=task_id, status=TaskStatus.PENDING, created_at=datetime.utcnow(), request_data=request_data ) self.tasks[task_id] = task # Запуск асинхронного выполнения asyncio.create_task(self._execute_task(task_id, request_data)) logger.info(f"✅ Task {task_id} created") return task_id
⚡ Метод _execute_task() с таймаутом
async def _execute_task(self, task_id: str, request_data: Dict): task = self.tasks[task_id] async with self._semaphore: # Захват семафора self._running_tasks += 1 task.status = TaskStatus.RUNNING task.started_at = datetime.utcnow() try: # Таймаут на выполнение timeout = self.task_timeout_minutes * 60 result = await asyncio.wait_for( self._run_llm_pipeline(request_data), timeout=timeout ) # Сохранение результата task.status = TaskStatus.COMPLETED task.result = result task.completed_at = datetime.utcnow() # Логирование I/O save_task_io(task_id, request_data, result) logger.info(f"✅ Task {task_id} completed") except asyncio.TimeoutError: task.status = TaskStatus.FAILED task.error = f"Timeout after {self.task_timeout_minutes}m" self.error_stats["timeout_errors"] += 1 logger.error(f"❌ Task {task_id} timeout") except Exception as e: task.status = TaskStatus.FAILED task.error = str(e) self._categorize_error(e) logger.exception(f"❌ Task {task_id} failed") finally: self._running_tasks -= 1

async def _cleanup_old_tasks(self): """Фоновая задача очистки старых завершенных задач""" while True: await asyncio.sleep(3600) # Каждый час cutoff = datetime.utcnow() - timedelta(hours=self.cleanup_after_hours) tasks_to_delete = [] for task_id, task in self.tasks.items(): # Удаляем только завершенные/ошибочные задачи if task.status in [TaskStatus.COMPLETED, TaskStatus.FAILED]: if task.completed_at and task.completed_at < cutoff: tasks_to_delete.append(task_id) # Удаление for task_id in tasks_to_delete: del self.tasks[task_id] logger.info(f"🗑️ Cleaned up task {task_id}") if tasks_to_delete: logger.info(f"🧹 Cleaned {len(tasks_to_delete)} old tasks") # Запуск при старте приложения @app.on_event("startup") async def startup_event(): task_manager.start_cleanup_task() logger.info("✅ Cleanup task started")

def _categorize_error(self, error: Exception): """Классификация ошибок для статистики""" error_msg = str(error).lower() if "llm" in error_msg or "openai" in error_msg: self.error_stats["llm_errors"] += 1 elif "validation" in error_msg or "invalid" in error_msg: self.error_stats["validation_errors"] += 1 elif "database" in error_msg or "jdbc" in error_msg: self.error_stats["database_errors"] += 1 self.error_stats["total_errors"] += 1 # GET /health endpoint @app.get("/health") async def health_check(): stats = task_manager.get_stats() error_rate = stats["failed_tasks"] / stats["total_tasks"] if stats["total_tasks"] > 0 else 0 health_status = "healthy" if error_rate > 0.5: health_status = "critical" elif error_rate > 0.2: health_status = "degraded" return { "health": health_status, "errors": task_manager.error_stats }

import trino from urllib.parse import urlparse, parse_qs def _connect_trino(jdbc_url: str): # Парсинг JDBC URL url = jdbc_url.replace('jdbc:trino://', 'http://') parsed = urlparse(url) params = parse_qs(parsed.query) # Параметры подключения connection = trino.dbapi.connect( host=parsed.hostname, port=parsed.port or 8080, user=params.get('user', ['admin'])[0], catalog=params.get('catalog', ['hive'])[0], schema=params.get('schema', ['default'])[0] ) logger.info(f"✅ Connected to Trino: {parsed.hostname}:{parsed.port}") return connection # Получение статистики таблицы def get_table_stats_trino(conn, catalog, schema, table): cursor = conn.cursor() # 1. Row count cursor.execute(f"SELECT COUNT(*) FROM {catalog}.{schema}.{table}") row_count = cursor.fetchone()[0] # 2. Table size (Iceberg specific) cursor.execute(f""" SELECT SUM(\"$data_size_in_bytes\") FROM \"{catalog}\".\"{schema}\".\"{table}$files\" """) size_bytes = cursor.fetchone()[0] or 0 # 3. Column cardinality (approximate for speed) cursor.execute(f"SHOW COLUMNS FROM {catalog}.{schema}.{table}") columns = [row[0] for row in cursor.fetchall()] column_stats = {} for col in columns: cursor.execute(f""" SELECT APPROX_DISTINCT({col}) FROM {catalog}.{schema}.{table} """) cardinality = cursor.fetchone()[0] column_stats[col] = {'cardinality': cardinality} return { 'row_count': row_count, 'size_bytes': size_bytes, 'columns': column_stats }
🐘 Подключение к PostgreSQL
import psycopg2 def _connect_postgresql(jdbc_url: str): # Конвертация JDBC URL в PostgreSQL connection string pg_url = jdbc_url.replace('jdbc:postgresql://', 'postgresql://') connection = psycopg2.connect(pg_url) logger.info("✅ Connected to PostgreSQL") return connection # Статистика через pg_stat def get_table_stats_postgres(conn, schema, table): cursor = conn.cursor() # 1. Row count и size из pg_stat cursor.execute(f""" SELECT reltuples::BIGINT as row_count, pg_total_relation_size('{schema}.{table}')::BIGINT as size_bytes FROM pg_class c JOIN pg_namespace n ON n.oid = c.relnamespace WHERE c.relname = '{table}' AND n.nspname = '{schema}' """) row_count, size_bytes = cursor.fetchone() # 2. Column statistics cursor.execute(f""" SELECT attname, n_distinct, null_frac FROM pg_stats WHERE schemaname = '{schema}' AND tablename = '{table}' """) column_stats = {} for col_name, n_distinct, null_frac in cursor.fetchall(): column_stats[col_name] = { 'cardinality': abs(int(n_distinct * row_count)) if n_distinct < 0 else int(n_distinct), 'null_percent': null_frac * 100 } return { 'row_count': row_count, 'size_bytes': size_bytes, 'columns': column_stats }

import mysql.connector def _connect_mysql(jdbc_url: str): parsed = urlparse(jdbc_url.replace('jdbc:mysql://', 'http://')) params = parse_qs(parsed.query) connection = mysql.connector.connect( host=parsed.hostname, port=parsed.port or 3306, user=params.get('user', ['root'])[0], password=params.get('password', [''])[0], database=parsed.path.lstrip('/') ) logger.info("✅ Connected to MySQL") return connection # Статистика через INFORMATION_SCHEMA def get_table_stats_mysql(conn, database, table): cursor = conn.cursor() # 1. Table size cursor.execute(f""" SELECT TABLE_ROWS, DATA_LENGTH + INDEX_LENGTH as size_bytes FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_SCHEMA = '{database}' AND TABLE_NAME = '{table}' """) row_count, size_bytes = cursor.fetchone() # 2. Column cardinality (approximate) cursor.execute(f""" SELECT COLUMN_NAME, CARDINALITY FROM INFORMATION_SCHEMA.STATISTICS WHERE TABLE_SCHEMA = '{database}' AND TABLE_NAME = '{table}' GROUP BY COLUMN_NAME """) column_stats = {} for col, card in cursor.fetchall(): column_stats[col] = {'cardinality': card or 0} return { 'row_count': row_count, 'size_bytes': size_bytes, 'columns': column_stats }

🔄 Retry логика с экспоненциальным backoff
def _call_with_retries(self, system_prompt, user_input, max_retries=3): """LLM вызов с повторными попытками""" for attempt in range(max_retries): try: response = self.client.chat.completions.create( model=self.analysis_model, messages=[ {"role": "system", "content": system_prompt}, {"role": "user", "content": user_input} ], temperature=0.1, max_tokens=2000 ) # Извлечение и парсинг JSON content = response.choices[0].message.content json_str = self._extract_json(content) result = json.loads(json_str) logger.info(f"✅ LLM call successful (attempt {attempt+1})") return result except json.JSONDecodeError as e: logger.warning(f"⚠️ JSON parse error (attempt {attempt+1}): {e}") if attempt == max_retries - 1: raise LLMAnalyzerError("Failed to parse LLM response") time.sleep(2 ** attempt) # Exponential backoff except Exception as e: logger.error(f"❌ LLM call failed (attempt {attempt+1}): {e}") if attempt == max_retries - 1: raise time.sleep(2 ** attempt)

# llm_analyzer.py - Генерация DDL def _generate_optimized_ddl( table_analysis: Dict, original_ddl: str, catalog_name: str, schema_name: str ) -> str: """ Генерация оптимизированного DDL на основе: 1. LLM анализа (партиции, кластеры) 2. Валидного SQL шаблона 3. Реальных колонок из original_ddl """ # Шаг 1: Парсинг оригинального DDL через SQLglot parsed = sqlglot.parse_one(original_ddl, dialect="trino") table_name = parsed.find(sqlglot.exp.Table).name # Шаг 2: Извлечение колонок columns = [] for col_def in parsed.find_all(sqlglot.exp.ColumnDef): columns.append({ 'name': col_def.name, 'type': col_def.kind.sql() }) # Шаг 3: Формирование columns SQL columns_sql = ",\n ".join([ f"{col['name']} {col['type']}" for col in columns ]) # Шаг 4: Получение рекомендаций LLM partition_cols = table_analysis.get('partition_columns', []) cluster_cols = table_analysis.get('cluster_columns', []) # Шаг 5: Построение WITH clause with_parts = [ "format = 'ICEBERG'", ] if partition_cols: parts_str = ", ".join([f"'{c}'" for c in partition_cols]) with_parts.append(f"partitioning = ARRAY[{parts_str}]") if cluster_cols: cluster_str = ", ".join([f"'{c}'" for c in cluster_cols]) with_parts.append(f"sorted_by = ARRAY[{cluster_str}]") # Добавление оптимизаций with_parts.extend([ "'write.compression-codec' = 'ZSTD'", "'write.target-file-size-bytes' = '268435456'", # 256MB "'read.vectorization.enabled' = 'true'", "'write.parquet.compression-codec' = 'ZSTD'", "'write.parquet.page-size-bytes' = '1048576'", # 1MB "'write.parquet.row-group-size-bytes' = '134217728'" # 128MB ]) with_clause = ",\n ".join(with_parts) # Шаг 6: Финальный DDL ddl = f"""CREATE TABLE {catalog_name}.{schema_name}.{table_name} ( {columns_sql} ) WITH ( {with_clause} )""" return ddl

import sqlglot from sqlglot import exp def optimize_query(query: str, catalog: str, schema: str) -> str: """Оптимизация SQL запроса через AST трансформации""" # 1. Парсинг SQL в AST parsed = sqlglot.parse_one(query, dialect="trino") # 2. Изменение путей таблиц на оптимизированные for table in parsed.find_all(exp.Table): table.set("catalog", exp.Identifier(this=catalog)) table.set("db", exp.Identifier(this=schema)) # 3. Замена SELECT * на явные колонки if _has_select_star(parsed): explicit_columns = _get_table_columns(catalog, schema, table_name) _replace_select_star(parsed, explicit_columns) # 4. Добавление LIMIT если нет агрегаций if not parsed.find(exp.Limit): if not _is_aggregation(parsed): parsed = parsed.limit(10000) logger.info("✅ Added LIMIT 10000") # 5. Проверка partition pruning where_clause = parsed.find(exp.Where) if where_clause: where_cols = {col.name for col in where_clause.find_all(exp.Column)} if 'created_at' in where_cols: logger.info("✅ Partition pruning on created_at") # 6. Генерация оптимизированного SQL optimized_sql = parsed.sql(dialect="trino") return optimized_sql

🗂️ Проверка partition pruning
def check_partition_pruning(parsed, partition_cols: List[str]) -> Dict: """Проверка использования партиционных колонок в WHERE""" where = parsed.find(sqlglot.exp.Where) if not where: return {"pruning": False, "reason": "No WHERE clause"} # Извлечение всех колонок из WHERE where_columns = { col.name for col in where.find_all(sqlglot.exp.Column) } # Проверка пересечения с партициями used_partitions = where_columns.intersection(set(partition_cols)) if used_partitions: logger.info(f"✅ Partition pruning on: {used_partitions}") return { "pruning": True, "columns": list(used_partitions), "benefit": "Reduced data scan" } else: logger.warning("⚠️ Full table scan - no partition pruning") return {"pruning": False, "reason": "No partition columns in WHERE"}
✅ Пример эффективного partition pruning:
